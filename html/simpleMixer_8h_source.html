45class SimpleMixer
46{
47private:
48 static constexpr int_least32_t SCALE_FACTOR = 1 << 16;
49
50#ifdef __cpp_lib_math_constants
51 static constexpr double SQRT_2 = std::numbers::sqrt2;
52 static constexpr double SQRT_3 = std::numbers::sqrt3;
53#else
54 static constexpr double SQRT_2 = 1.41421356237;
55 static constexpr double SQRT_3 = 1.73205080757;
56#endif
57
58 static constexpr int_least32_t SCALE[3] = {
59 SCALE_FACTOR, // 1 chip, no scale
60 static_cast<int_least32_t>((1.0 / SQRT_2) * SCALE_FACTOR), // 2 chips, scale by sqrt(2)
61 static_cast<int_least32_t>((1.0 / SQRT_3) * SCALE_FACTOR) // 3 chips, scale by sqrt(3)
62 };
63
64private:
65 using mixer_func_t = int_least32_t (SimpleMixer::*)() const;
66
67private:
68 std::vector<short*> m_buffers;
69
70 std::vector<int_least32_t> m_iSamples;
71
72 std::vector<mixer_func_t> m_mix;
73
74private:
75 void updateParams();
76
77 /*
78 * Channel matrix
79 *
80 * C1
81 * L 1.0
82 * R 1.0
83 *
84 * C1 C2
85 * L 1.0 0.5
86 * R 0.5 1.0
87 *
88 * C1 C2 C3
89 * L 1.0 1.0 0.5
90 * R 0.5 1.0 1.0
91 */
92
93 // Mono mixing
94 template <int Chips>
95 int_least32_t mono() const
96 {
97 int_least32_t res = 0;
98 for (int i = 0; i < Chips; i++)
99 res += m_iSamples[i];
100 return res * SCALE[Chips-1] / SCALE_FACTOR;
101 }
102
103 // Stereo mixing
104 int_least32_t stereo_OneChip() const { return m_iSamples[0]; }
105
106 int_least32_t stereo_ch1_TwoChips() const
107 {
108 return (m_iSamples[0] + 0.5*m_iSamples[1]) * SCALE[1] / SCALE_FACTOR;
109 }
110 int_least32_t stereo_ch2_TwoChips() const
111 {
112 return (0.5*m_iSamples[0] + m_iSamples[1]) * SCALE[1] / SCALE_FACTOR;
113 }
114
115 int_least32_t stereo_ch1_ThreeChips() const
116 {
117 return (m_iSamples[0] + m_iSamples[1] + 0.5*m_iSamples[2]) * SCALE[2] / SCALE_FACTOR;
118 }
119 int_least32_t stereo_ch2_ThreeChips() const
120 {
121 return (0.5*m_iSamples[0] + m_iSamples[1] + m_iSamples[2]) * SCALE[2] / SCALE_FACTOR;
122 }
123
124private:
125 SimpleMixer() = delete;
126 SimpleMixer(const SimpleMixer&) = delete;
127
128public:
132 SimpleMixer(bool stereo, short** buffers, int chips);
133
137 unsigned int doMix(short *buffer, unsigned int samples);
138};
