78  // find the right spot where to tuck this new event
79  Event **scan = &firstEvent;
80  for (;;)
81  {
82  if ((*scan == nullptr) || ((*scan)->triggerTime > event.triggerTime))
83  {
84  event.next = *scan;
85  *scan = &event;
86  break;
87  }
88  scan = &((*scan)->next);
89  }
