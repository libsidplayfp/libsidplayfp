59  bool input(int sample) override

63  if (sampleOffset < 1024)
64  {
65  outputValue = cachedSample + (sampleOffset * (sample - cachedSample) >> 10);
66  ready = true;
67  sampleOffset += cyclesPerSample;
68  }
69 
70  sampleOffset -= 1024;
71

73 
74  return ready;
75  }
