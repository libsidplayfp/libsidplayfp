113int ExternalFilter::clock(int input)
114{
115 const int Vi = input << 11;
116 const int dVlp = (w0lp_1_s7 * (Vi - Vlp) >> 7);
117 const int dVhp = (w0hp_1_s17 * (Vlp - Vhp) >> 17);
118 Vlp += dVlp;
119 Vhp += dVhp;
120 return (Vlp - Vhp) >> 11;
121}
